var SS=SpreadsheetApp.getActiveSpreadsheet(); function getOrCreate(n){return SS.getSheetByName(n)||SS.insertSheet(n);} function doGet(e){ var t=(e.parameter&&e.parameter.type)||"products"; var sh=getOrCreate(t==="settings"?"Settings":"Products"); var v=sh.getRange("A1").getValue(); if(!v)v=t==="settings"?"{}":"[]"; return ContentService.createTextOutput(v).setMimeType(ContentService.MimeType.JSON); } function doPost(e){ try{ var d=JSON.parse(e.postData.contents); if(d.type==="products"){getOrCreate("Products").getRange("A1").setValue(JSON.stringify(d.data));return ContentService.createTextOutput("OK");} if(d.type==="settings"){getOrCreate("Settings").getRange("A1").setValue(JSON.stringify(d.data));return ContentService.createTextOutput("OK");} var s=getOrCreate("Orders"); if(s.getLastRow()===0)s.appendRow(["ID","Time","Name","Phone","City","Address","Items","Total","Payment","Notes"]); s.appendRow([d.id,d.time,d.name,d.phone,d.city,d.address,d.items,d.total,d.payment,d.notes||""]); return ContentService.createTextOutput("OK"); }catch(err){return ContentService.createTextOutput("ERR:"+err);} }
